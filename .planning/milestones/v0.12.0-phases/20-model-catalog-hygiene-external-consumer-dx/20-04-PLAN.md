```ts
import type { Registry } from './registry.js';

export interface DeprecationMeta {
  old_name: string;
  new_name: string;
  deprecated_since: string;
  removal_target: string;
}

export type DeprecationMap = Record<string, {
  target: string;
  deprecated_since: string;
  removal_target: string;
}>;

export interface ResolveAliasResult {
  /** The canonical alias to actually dispatch against. Equals input when no deprecation entry hits. */
  canonical: string;
  /** Populated only when the input was a deprecated alias. */
  deprecation_meta: DeprecationMeta | undefined;
}

/**
 * Phase 20 (v0.12.0 — CAT-04 / D-02 + D-03): if `alias` is in the operator-declared
 * deprecation map, return the canonical target plus metadata for header/log/metric.
 * Otherwise pass through.
 *
 * Composes with Wave 0 (CAT-01): a disabled entry whose name is ALSO in the
 * deprecation map will resolve at dispatch time even though it's invisible at
 * the /v1/models surface. This asymmetry is intentional and documented in
 * 20-CONTEXT.md §6.
 */
export function resolveAlias(alias: string, registry: Registry): ResolveAliasResult {
  const dep = registry.deprecated_aliases?.[alias];
  if (!dep) return { canonical: alias, deprecation_meta: undefined };
  return {
    canonical: dep.target,
    deprecation_meta: {
      old_name: alias,
      new_name: dep.target,
      deprecated_since: dep.deprecated_since,
      removal_target: dep.removal_target,
    },
  };
}
```

```ts
import { resolveAlias } from '../config/deprecation.js';
import type { DeprecationMeta } from '../config/deprecation.js';

export async function applyPreflight(
  requested_model: string,
  opts: ApplyPreflightOpts,
): Promise<ApplyPreflightResult> {
  const snapshot = opts.registry.get();

  // Phase 20 (CAT-04): deprecation map intercepts BEFORE registry.resolve().
  // If the input is a deprecated alias, the canonical target is what we actually
  // dispatch against; the deprecation_meta rides through to the route handler.
  const aliasResult = resolveAlias(requested_model, snapshot);
  const canonical = aliasResult.canonical;

  // registry.resolve(canonical) — note: this enforces the Wave 0 contract that
  // disabled entries are unresolvable, BUT only when the canonical target is
  // disabled (which would be a config error caught by Task 1's superRefine).
  // A deprecated->enabled mapping resolves cleanly.
  const entry = opts.registry.resolve(canonical);

  // Policy gate runs against the canonical target's policy block (NOT the
  // deprecated alias's — the deprecated entry's policy is moot once redirected).
  applyPolicyGate(snapshot.policies, entry, canonical);

  const breakerResult = await opts.breaker.check(entry.backend);
  return {
    entry,
    breakerState: breakerResult.state,
    deprecation_meta: aliasResult.deprecation_meta,
  };
}
```

POL-05 invariant preserved: the policy gate still runs BEFORE the breaker check. The only behavior change is alias resolution at the front of the pipeline.
