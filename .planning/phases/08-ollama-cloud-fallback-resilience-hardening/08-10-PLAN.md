```bash
# bin/smoke-test-cloud.sh — Phase 8 smoke (live).

#   8. cloud_spend_daily view exists (CLOUD-05)

# ... color helpers, PASS/FAIL tracking, ROUTER_BEARER_TOKEN load from .env ...

# curl -I against /v1/models with bearer; grep for x-model-backend on /v1/chat/completions, /v1/messages, /v1/embeddings response.
# Expected: header present with valid backend value.

echo "=== Section 2: Cloud chat completions live (CLOUD-01, CLOUD-02) ==="
if [[ -z "${OLLAMA_API_KEY:-}" ]]; then

else
  # curl POST /v1/chat/completions -d '{"model":"gpt-oss:20b-cloud","messages":[{"role":"user","content":"reply with the word OK and nothing else"}],"stream":false}'
  # Assert: HTTP 200; response JSON has .choices[0].message.content containing "OK"; X-Model-Backend: ollama-cloud.
fi

echo "=== Section 3: Cloud embeddings live (EMBED-02) ==="
# Similar — only if OLLAMA_API_KEY set AND a cloud embedding model is declared.
# Smoke MAY skip this section if no cloud embedding entry exists; operator
# can add one to models.yaml to exercise.

echo "=== Section 4: max_tokens cap on cloud → 400 (CLOUD-04) ==="
# curl POST /v1/chat/completions -d '{"model":"gpt-oss:20b-cloud","messages":[...],"max_tokens":32768}'
# Assert: HTTP 400; envelope code='cloud_max_tokens_exceeded'.
# Adapter never called (no upstream traffic — verify via Valkey breaker state unchanged).

echo "=== Section 5: Circuit breaker via Valkey direct manipulation (CLOUD-03) ==="
# 1. valkey-cli SET breaker:ollama-cloud:state open EX 60
# 2. valkey-cli SET breaker:ollama-cloud:probe_at $(( $(date +%s)*1000 + 30000 )) EX 60
# 3. curl POST /v1/chat/completions with cloud model → expect 503 + envelope code='backend_circuit_open' + Retry-After: 60.
# 4. valkey-cli DEL breaker:ollama-cloud:state breaker:ollama-cloud:probe_at  # reset
# 5. curl again → expect 200 (or 502 if cloud unreachable; either is "non-503" which is the goal).

echo "=== Section 6: Rate limit 429 via burst (ROUTE-11) ==="
# 1. Set ROUTER_RATE_LIMIT_RPM=5 (or use default 600 if too slow to test 600 requests):
#    The smoke uses a small RPM via a temporary env override OR by checking the first N requests.
# 2. fire 7 concurrent (or sequential) /v1/chat/completions requests with same bearer.
# 3. Assert: requests 1-5 (or 1-N where N = RPM) return 200; the (N+1)th returns 429 + envelope code='rate_limit_exceeded' + Retry-After: 60.
# NOTE: If the default RPM=600 is in effect, the smoke fires 605 requests; takes ~30s; acceptable.
# Alternative: prefix the section with a "skip if RPM > 50" check to keep the smoke fast.

echo "=== Section 7: Idempotency mux dedup via concurrent same-key (ROUTE-12) ==="
# 1. KEY=$(date +%s%N)-$(uuidgen | head -c 8)
# 2. fire 3 concurrent POST /v1/chat/completions with Idempotency-Key: $KEY (same prompt, non-stream).
# 3. Assert: all 3 return HTTP 200; the response bodies are byte-identical (md5sum compare).
# 4. Verify in request_log: 3 rows with the same upstream_message_id (psql query).
```

```markdown
## Phase 8 — Cloud fallback + resilience

### Bring up Valkey
The Valkey service (Plan 08-01) runs on the `data` network and is required for breaker, rate-limit, idempotency, and registry-cache features.

  docker compose up -d valkey postgres
  docker compose ps valkey  # verify healthy

### Cloud models (CLOUD-01 / CLOUD-02 / EMBED-02)
1. Get an Ollama Cloud API key: https://ollama.com → Settings → API Keys → Create.
2. Set `OLLAMA_API_KEY=oss_...` in .env.
3. Verify cloud entries exist in router/models.yaml (Plan 08-02 added gpt-oss:120b-cloud + gpt-oss:20b-cloud).
4. Restart router: `docker compose restart router`.
5. Smoke: `curl -H "Authorization: Bearer $ROUTER_BEARER_TOKEN" http://localhost:3000/v1/chat/completions -d '{"model":"gpt-oss:20b-cloud","messages":[{"role":"user","content":"hi"}],"stream":false}' | jq`.

### X-Model-Backend header (ROUTE-10)
Every response carries `X-Model-Backend: <backend>` (verify with `curl -I` against /v1/models or any chat/messages/embeddings endpoint).

### Circuit breaker (CLOUD-03)
Inspect state:
  docker compose exec -T valkey valkey-cli -a $VALKEY_PASSWORD KEYS 'breaker:*'
Reset manually (during incident):
  docker compose exec -T valkey valkey-cli -a $VALKEY_PASSWORD DEL breaker:ollama-cloud:state breaker:ollama-cloud:fail_count breaker:ollama-cloud:probe_at

### max_tokens cap (CLOUD-04)
Cloud models reject `max_tokens > 16384` with a structured 400. Verify:
  curl ... -d '{"model":"gpt-oss:20b-cloud","max_tokens":32768,...}'

### Rate limit (ROUTE-11)
Inspect counters:
  docker compose exec -T valkey valkey-cli -a $VALKEY_PASSWORD KEYS 'ratelimit:*'

### Idempotency (ROUTE-12)
Recipe for retry-safe agent code:
  KEY=$(uuidgen)
  curl -H "Idempotency-Key: $KEY" ... /v1/chat/completions
  # If the request fails, retry with the SAME $KEY — the router multiplexes.

### cloud_spend_daily (CLOUD-05)
  docker compose exec -T postgres psql -U app -d router -c "SELECT * FROM cloud_spend_daily;"

### Phase 8 smoke
  bash bin/smoke-test-cloud.sh
```
