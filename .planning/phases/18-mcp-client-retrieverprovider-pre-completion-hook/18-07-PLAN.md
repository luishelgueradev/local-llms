```typescript
/**
 * Phase 18 (v0.11.0 — MCPC-01..06 + RETR-02..06)
 *
 * Shared helper used by /v1/chat/completions, /v1/messages, and /v1/responses
 * to implement the "pre-completion hook chain + MCP external tool injection"
 * block in a single insertion shape (PATTERNS line 323 — single change shape
 * repeated three times; mirrors Phase 17 helpers/session-attach.ts).
 *
 * Inserted in each route IMMEDIATELY after the Phase 17 session-attach block,
 * BEFORE the *RequestToCanonical(...) call so the fenced content lands in
 * canonical.system (NOT canonical.messages — CTXP-03 BLOCK invariant).
 *
 * STATELESS + SIDE-EFFECT-FREE wrt to its inputs except for:
 *   - reply.header('X-Hook-Error', ...) on fail-open (single header stamp; Pitfall 17-D analog)
 *   - (req as { hookLog }).hookLog = hook_log  (stashed for recordOutcome → request_log.hook_log)

import type { FastifyReply, FastifyRequest } from 'fastify';
import type { CanonicalRequest, CanonicalTool } from '../../../translation/canonical.js';
import type { ModelEntry } from '../../../config/registry.js';
import type { McpClientRegistry } from '../../../mcp/client/registry.js';
import type {
  PreCompletionHook,
  HookLogEntry,
} from '../../../hooks/pre-completion.js';
import { runHookChain } from '../../../hooks/pre-completion.js';
import type { MetricsRegistry } from '../../../metrics/registry.js';

export type Phase18RouteKey =
  | '/v1/chat/completions'
  | '/v1/messages'
  | '/v1/responses';

export interface RunPreCompletionAndInjectMcpToolsOpts {
  routeKey: Phase18RouteKey;
  preCompletionHooks?: Map<string, PreCompletionHook[]>;
  mcpClientRegistry?: McpClientRegistry;
  metrics: MetricsRegistry;
}

export interface RunPreCompletionAndInjectMcpToolsResult {
  /** Possibly-mutated canonical: system field appended with fenced content + tools appended. */
  canonical: CanonicalRequest;
  /** Hook audit entries — stashed on req.hookLog AND returned for explicit handler use. */
  hook_log: HookLogEntry[];
  /**
   * True when:
   *   - entry.mcp_servers_enabled has length AND
   *   - opts.mcpClientRegistry is present AND
   *   - canonical.stream is falsy (Phase 18 RESOLVED #4 — non-stream only).
   *
   * The route handler MUST wrap its adapter call in runMcpToolLoop() when this is true.
   */
  mcpToolLoopEnabled: boolean;
}

export async function runPreCompletionAndInjectMcpTools(
  req: FastifyRequest,
  reply: FastifyReply,
  canonical: CanonicalRequest,
  entry: ModelEntry,
  opts: RunPreCompletionAndInjectMcpToolsOpts,
): Promise<RunPreCompletionAndInjectMcpToolsResult> {
  // ── 1. Pre-completion hook chain (RETR-02 — fires AFTER ContextProvider, BEFORE adapter) ──
  const hooks = opts.preCompletionHooks?.get(opts.routeKey) ?? [];
  const {
    canonical: c1,
    hook_log,
    fail_open_signaled,
    fail_open_hook_name,
  } = await runHookChain(req, canonical, hooks, opts.metrics);

  if (fail_open_signaled && fail_open_hook_name) {
    // Pitfall 17-D analog: stamp X-Hook-Error BEFORE reply.send/reply.sse (Fastify v5 onSend
    // fires sync inside reply.send; setting headers after .send is a no-op).
    // RESOLVED #7 format: `<hook_name>:<reason>` literal.
    void reply.header('X-Hook-Error', `${fail_open_hook_name}:timeout`);
  }

  // ── 2. MCP external tool injection (MCPC-03 — namespace-prefixed) ──
  let c2 = c1;
  let mcpToolLoopEnabled = false;
  const enabledAliases = (entry as ModelEntry & { mcp_servers_enabled?: string[] }).mcp_servers_enabled ?? [];

  if (
    enabledAliases.length > 0 &&
    opts.mcpClientRegistry &&
    !c2.stream
  ) {
    const injectedTools: CanonicalTool[] = [];
    for (const alias of enabledAliases) {
      try {
        const tools = await opts.mcpClientRegistry.getOrFetchTools(alias);
        injectedTools.push(...tools);
      } catch (err) {
        // P2-01 BLOCK: external MCP server unreachable at request time → log + skip.
        // We do NOT fail the request — the model can still proceed without the tools.
        // Plan 18-04's registry already throws McpServerUnreachableError; we catch here
        // to keep this as a non-fatal degradation (different from fail-closed on hook).
        req.log.warn(
          { alias, err: String(err), event: 'mcp_tools_fetch_failed' },
          `MCP tools/list failed for alias "${alias}" — proceeding without its tools`,
        );
      }
    }
    if (injectedTools.length > 0) {
      // RESOLVED #10: APPEND, do NOT replace. User-provided tools have priority (first in array).
      c2 = { ...c2, tools: [...(c2.tools ?? []), ...injectedTools] };
      mcpToolLoopEnabled = true;
    }
  }

  // ── 3. Stash hook_log on req for recordOutcome (writes to request_log.hook_log JSONB) ──
  (req as FastifyRequest & { hookLog?: HookLogEntry[] }).hookLog = hook_log;

  return { canonical: c2, hook_log, mcpToolLoopEnabled };
}
```
