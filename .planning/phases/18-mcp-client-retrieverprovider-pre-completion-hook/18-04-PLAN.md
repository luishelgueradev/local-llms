```typescript
/**
 * Phase 18 (v0.11.0 — P2-04 BLOCK — outbound MCP auth isolation)
 *
 * THIS FILE IS THE ONLY P2-04 BOUNDARY:
 *
 *   - `buildOutboundHeaders(cfg)` takes ONLY the McpServerConfig — NEVER a
 *     FastifyRequest or any header bag from the inbound request.
 *   - `StreamableHTTPClientTransport`'s `requestInit.headers` is the ONLY
 *     auth surface the SDK exposes.
 *
 * Grep gate enforced by Plan 18-01 tests:
 *   grep -rE "req\.headers|request\.headers" router/src/mcp/client/
 *   MUST return empty.
 */

import { Client } from '@modelcontextprotocol/sdk/client/index.js';
import { StreamableHTTPClientTransport } from '@modelcontextprotocol/sdk/client/streamableHttp.js';
import type { McpServerConfig } from './registry.js';

/**
 * Build the outbound auth headers from server config alone.
 *
 * STRICT INPUTS: ONLY the McpServerConfig. The function signature does NOT
 * accept a FastifyRequest, headers bag, or any context-bearing object.
 *
 * @returns headers map (possibly empty for auth_type='none').
 */
export function buildOutboundHeaders(cfg: McpServerConfig): Record<string, string> {
  if (cfg.auth_type === 'bearer' && cfg.auth_value) {
    return { Authorization: `Bearer ${cfg.auth_value}` };
  }
  // auth_type === 'none' OR (bearer + missing auth_value): empty headers.
  // The latter is a config error caught at registry-construction time (Zod superRefine).
  return {};
}

/**
 * Construct a fresh Client + transport pair for the given config.
 *
 * Caller is responsible for calling `client.connect(transport)` after construction
 * (lazy-connect pattern — registry.ts does this on first getOrConnect).
 *
 * @param cfg The server config.
 * @returns   { client, transport } — caller wires lifecycle.
 */
export function buildClient(cfg: McpServerConfig): {
  client: Client;
  transport: StreamableHTTPClientTransport;
} {
  const client = new Client({
    name: 'local-llms-router',
    version: '0.11.0',
  });

  const transport = new StreamableHTTPClientTransport(new URL(cfg.url), {
    requestInit: {
      headers: buildOutboundHeaders(cfg),
    },
  });

  return { client, transport };
}
```

   - Private `connections: Map<string, Promise<ConnectedEntry>>` (NOT `Map<string, ConnectedEntry>` — store the in-flight promise to coalesce concurrent connect calls per-alias).
   - Constructor takes `MakeMcpClientRegistryOpts`, stores them. **NO `connect()` call here** (P2-01 BLOCK).

4. **`getOrConnect(alias)`:**
   - Look up cached promise; if absent, call `this.connectOne(alias)`, store the promise, and add `.catch(() => this.connections.delete(alias))` to allow retry on next call.
   - Return `(await p).client`.

5. **`connectOne(alias)`:**
   - Look up `cfg = this.opts.servers.get(alias)`. If absent, throw `Error("MCP server alias not configured: ...")` (NOT McpServerUnreachableError — this is a programmer error).
   - Call `buildClient(cfg)` (from transport.ts) to get `{ client, transport }`.
   - Wrap `await client.connect(transport)` in try/catch — on throw, wrap as `new McpServerUnreachableError(alias, cfg.url, err)` and rethrow.
   - Return `{ client, transport, config: cfg }`.

6. **`getOrFetchTools(alias)`:**
   - Cache key: `mcp:tools:${alias}`.
   - If `this.opts.valkey` present, `GET` the key; if hit, parse and return `tools.map(t => ({ ...t, name: prefixToolName(alias, t.name) }))`. **Tools cached are ALREADY SANITIZED (no prefix yet — we add prefix on every read so the cached form is prefix-free + cheap)**.
   - On cache miss: `await this.getOrConnect(alias)`, then `await client.listTools()`, then map each raw tool through `sanitizeExternalTool(t, alias, this.opts.logger)` and `.filter((t): t is SanitizedTool => t !== null)`. Optionally apply `cfg.tool_filter` — if `['*']` (default), all; else filter by exact name match.
   - Store the sanitized + filtered list in Valkey with `EX (this.opts.cacheTtlSec ?? 60)`.
   - Return prefixed CanonicalTool array.

7. **`callTool(alias, toolName, args)`:**
   - `await this.getOrConnect(alias)`.
   - `return await client.callTool({ name: toolName, arguments: args as Record<string, unknown> }, undefined, { timeout: cfg.timeout_ms ?? 10_000 })`.

8. **`dispose(alias)`:**
   - Pop from `connections` Map.
   - If Valkey present, `DEL mcp:tools:{alias}`.
   - If a connection existed: `await (await p).transport.close()` inside try/catch (swallow errors on dispose).

9. **`disposeAll()`:**
   - Capture keys: `const aliases = [...this.connections.keys()];`.
   - `await Promise.all(aliases.map(a => this.dispose(a)));`.
   - **Add 5-second Promise.race timeout** (mirrors Phase 15 `shutdownSessions` pattern in `session-gc.ts`): wrap each dispose() in `Promise.race([dispose, timeout(5000)])` so a wedged transport.close() cannot block SIGTERM indefinitely.

10. **Factory `makeMcpClientRegistry(opts)`:** simple constructor wrapper. Returns the class instance typed as `McpClientRegistry`.
