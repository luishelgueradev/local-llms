**File 1: `router/src/mcp/client/tool-loop.ts` (NEW)**

```typescript
/**
 * Phase 18 (v0.11.0 — MCPC-04)
 *
 * runMcpToolLoop: drive the model→external-MCP-tool→model loop with a hard
 * 10-iteration cap. Per-request, sequential between iterations; tool calls
 * within one iteration run in parallel (Promise.all).
 *
 * SCOPE (RESOLVED Open Question #4): NON-STREAM paths only for Phase 18.
 * The stream path keeps Phase 16 behavior unchanged (the model emits tool
 * call events but the router does NOT loop on them — RESS-FUT).
 *
 * Protocol-agnostic: NO imports from router/src/routes/ or router/src/mcp/host/.
 * Mirrors the `dispatch/preflight.ts` helper-isolation invariant.
 */

import type { CanonicalRequest, CanonicalResponse } from '../../translation/canonical.js';
import type { BackendAdapter } from '../../backends/adapter.js';
import type { Logger } from 'pino';
import type { Counter } from 'prom-client';
import type { McpClientRegistry } from './registry.js';
import { stripPrefix, isExternalMcpToolCall } from './prefix.js';
import { McpToolLoopExceededError } from '../../errors/envelope.js';

/** MCPC-04 hard cap. REQUIREMENTS.md line 84 + ROADMAP.md line 216 say 10 (A8 lock). */
export const MCP_TOOL_LOOP_MAX = 10;

export interface RunMcpToolLoopOpts {
  initial: CanonicalRequest;
  adapter: BackendAdapter;
  signal: AbortSignal;
  registry: McpClientRegistry;
  enabledAliases: readonly string[];
  log: Logger;
  metrics: {
    routerMcpToolCallsExternalTotal: Counter<'server_alias' | 'status_class'>;
  };
}

/**
 * Drive the model→external-tool→model loop until:
 *   - Model returns no external tool calls → resolve with final response.
 *   - Iteration count reaches MCP_TOOL_LOOP_MAX (10) → throw McpToolLoopExceededError.
 *   - Abort signal fires → propagate (adapter / registry will throw).
 *
 * Internal MCP host tool-calls (no `{alias}__` prefix in the function name)
 * are IGNORED by this loop — they're handled by the existing adapter's
 * own tool-call path (or, for the host surface, by `mcp/host/tools/*` directly).
 */
export async function runMcpToolLoop(opts: RunMcpToolLoopOpts): Promise<CanonicalResponse> {
  let canonical = opts.initial;
  let iter = 0;

  // First adapter call — issued unconditionally.
  let resp = await opts.adapter.chatCompletionsCanonical(canonical, opts.signal);

  // Helper: is the response carrying an external MCP tool call?
  const hasExternalToolCall = (r: CanonicalResponse): boolean =>
    r.tool_calls?.some((tc) => isExternalMcpToolCall(tc.function.name, opts.enabledAliases)) ?? false;

  while (hasExternalToolCall(resp) && iter < MCP_TOOL_LOOP_MAX) {
    iter++;

    // Collect ONLY the external MCP tool calls — internal/host tool-calls pass through.
    const externalCalls = (resp.tool_calls ?? []).filter((tc) =>
      isExternalMcpToolCall(tc.function.name, opts.enabledAliases),
    );

    // Parallel dispatch within ONE iteration (per RESEARCH "Key invariants" line 527).
    const toolMessages = await Promise.all(
      externalCalls.map(async (tc) => {
        const stripped = stripPrefix(tc.function.name);
        // stripPrefix returns null only if `tc.function.name` lacks `__`; but isExternalMcpToolCall
        // already verified the prefix exists + matches an enabled alias, so this assertion is safe.
        if (!stripped) {
          throw new Error(`runMcpToolLoop invariant: isExternalMcpToolCall passed but stripPrefix returned null for "${tc.function.name}"`);
        }
        const { alias, toolName } = stripped;
        try {
          // Args are JSON-encoded per OpenAI tool-call spec; safely parse.
          const args = JSON.parse(tc.function.arguments);
          const result = await opts.registry.callTool(alias, toolName, args);
          opts.metrics.routerMcpToolCallsExternalTotal.inc({ server_alias: alias, status_class: 'success' });
          return {
            role: 'tool' as const,
            tool_call_id: tc.id,
            content: JSON.stringify(result),
          };
        } catch (err) {
          // Surface tool failure to the model as a tool_result with error payload —
          // RESEARCH §Pattern 4 lines 508-512. The model can then choose to retry or give up.
          opts.metrics.routerMcpToolCallsExternalTotal.inc({ server_alias: alias, status_class: 'server_error' });
          opts.log.warn(
            { alias, tool: toolName, err: String(err), event: 'mcp_tool_call_failed' },
            `external MCP tool call ${alias}__${toolName} failed`,
          );
          return {
            role: 'tool' as const,
            tool_call_id: tc.id,
            content: JSON.stringify({ error: String(err) }),
          };
        }
      }),
    );

    // Append the assistant message that triggered the calls + the tool results.
    // Use type-assertion bridges that match the canonical schema (the existing route handlers
    // do the same shape when threading tool messages — see chat-completions.ts pre-Phase-18
    // tool-call path for reference).
    const assistantMessage = {
      role: 'assistant' as const,
      content: resp.content ?? '',
      tool_calls: resp.tool_calls,
    };
    canonical = {
      ...canonical,
      messages: [...canonical.messages, assistantMessage as any, ...(toolMessages as any[])],
    };

    // Issue next adapter call. abort signal propagates per the BackendAdapter contract.
    resp = await opts.adapter.chatCompletionsCanonical(canonical, opts.signal);
  }

  if (iter >= MCP_TOOL_LOOP_MAX && hasExternalToolCall(resp)) {
    opts.log.warn(
      { iter, event: 'mcp_tool_loop_exceeded' },
      `MCP tool-call loop hit cap of ${MCP_TOOL_LOOP_MAX} iterations`,
    );
    throw new McpToolLoopExceededError(MCP_TOOL_LOOP_MAX);
  }

  return resp;
}
```

**Notes on implementation choices:**

- The assistant-message/tool-message shape uses `as any` casts at the canonical schema boundary because the canonical `CanonicalMessage` discriminated union has specific roles + variants; the bridge between BackendAdapter's response shape and the canonical messages array is well-trodden in the existing adapter/route code (Plan 18-06's runHookChain + Plan 18-07's helpers stay clean of this; this is a single-helper concern).
- If the existing adapter has typed message helpers (look for `appendToolMessages` or similar), prefer those over the type-assertion bridges.
- `JSON.parse(tc.function.arguments)` follows the OpenAI tool-call contract (string-encoded JSON); if the model emits malformed JSON, the catch block handles it as a tool-side error message back to the model (graceful recovery).
