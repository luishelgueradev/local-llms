const filterAndProject = (reg: ReturnType<RegistryStore['get']>) => {
  const allow = reg.policies?.default?.model_allowlist ?? [];
  return reg.models
    .filter((m) => allow.length === 0 || allow.includes(m.name))
    .map((m) => ({
      id: m.name, object: 'model' as const, created: registry.getCreatedAtSec(),
      owned_by: 'local-llms' as const, capabilities: m.capabilities,
      policy: { cloud_allowed: m.policy?.cloud_allowed ?? true },
    }));
};
