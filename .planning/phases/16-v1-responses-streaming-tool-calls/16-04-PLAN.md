/**
 * heartbeat-no-data-event.test.ts — Phase 16 (v0.11.0 — RESS / P3-04 BLOCK).
 *
 * Asserts that no file under router/src/ writes a `data:` event whose payload
 * contains the string "heartbeat". The SSE heartbeat MUST be a comment line
 * (`: keep-alive\n\n`) — see .planning/research/PITFALLS.md §"P3-04: Heartbeat
 * Collision With response.* Events" for the full justification.
 *
 * Mirrors the Phase 15 MCPS-06 stdio grep gate pattern at
 * router/tests/unit/mcp/host/stdio-grep-gate.test.ts.
 */
import { execSync } from 'node:child_process';
import { describe, it, expect } from 'vitest';
import path from 'node:path';

describe('P3-04 grep gate: heartbeat MUST be SSE comment line, not data event', () =&gt; {
  it('no router/src/ file writes a heartbeat as a data: event', () =&gt; {
    const srcDir = path.resolve(process.cwd(), 'src');
    // `|| true` so grep's "no match" exit-1 does not abort the test.
    const out = execSync(`grep -rnE "reply\\.raw\\.write.*heartbeat" ${srcDir} || true`, { encoding: 'utf8' });
    expect(out.trim()).toBe('');
  });

  it('no router/src/ file emits data: events containing the string "heartbeat"', () =&gt; {
    const srcDir = path.resolve(process.cwd(), 'src');
    const out = execSync(`grep -rnE "yield.*heartbeat|emit.*heartbeat" ${srcDir} || true`, { encoding: 'utf8' });
    expect(out.trim()).toBe('');
  });

  it('no router/src/translation/ file contains [DONE] terminator (Responses uses response.completed)', () =&gt; {
    // P3-03 / RESS-04 supplementary gate — the Responses API does NOT use [DONE].
    const translationDir = path.resolve(process.cwd(), 'src/translation');
    const out = execSync(`grep -rnF "[DONE]" ${translationDir}/responses-stream.ts || true`, { encoding: 'utf8' });
    expect(out.trim()).toBe('');
  });
});
```

**Anti-pattern guard:**
- DO NOT use `--exclude-dir=node_modules` — the grep is on `src/` only, never traverses node_modules.
- DO NOT remove the `|| true` suffix — grep returns exit 1 on "no match", which would falsely abort the test.
- DO NOT make the grep case-insensitive — exact-string matching is the contract; case-insensitivity would catch unrelated strings like "HeartBeat" in unrelated contexts.

# ============================================================================
# Phase 16 (v0.11.0 — RESS-01..05) — /v1/responses streaming + tool calls
# ============================================================================

echo ""
echo "[smoke-test-router] === Phase 16: /v1/responses streaming ==="

# Issue a streaming request and capture the SSE body.
RESS_TMP=$(mktemp /tmp/ress-smoke-XXXXXX.sse)
trap 'rm -f "$RESS_TMP"' EXIT
HTTP_CODE=$(curl -sN -o "$RESS_TMP" -w "%{http_code}" \
  -H "Authorization: Bearer ${ROUTER_BEARER_TOKEN}" \
  -H 'Content-Type: application/json' \
  -d "{\"model\":\"${MODEL}\",\"input\":\"say hi in 3 words\",\"stream\":true}" \
  "${ROUTER_URL}/v1/responses")

if [ "$HTTP_CODE" = "200" ]; then
  pass "RESS-01: POST /v1/responses stream:true returns 200"
else
  fail "RESS-01: POST /v1/responses stream:true returned ${HTTP_CODE} (expected 200)"
fi

# RESS-01: response.created event emitted.
if grep -q '^event: response.created' "$RESS_TMP"; then
  pass "RESS-01: response.created emitted"
else
  fail "RESS-01: missing response.created"
fi

# RESS-01 + P3-03: response.completed event emitted AND is the last event.
if grep -q '^event: response.completed' "$RESS_TMP"; then
  pass "RESS-01: response.completed emitted"
else
  fail "RESS-01: missing response.completed terminator"
fi

LAST_EVENT=$(grep '^event:' "$RESS_TMP" | tail -1)
if [ "$LAST_EVENT" = "event: response.completed" ]; then
  pass "RESS-02 / P3-03: response.completed is the last event (last-event invariant)"
else
  fail "RESS-02 / P3-03: last event is '${LAST_EVENT}', expected 'event: response.completed'"
fi

# RESS-02: every event payload carries sequence_number (sample first 3 events).
SEQ_COUNT=$(grep -cE '"sequence_number":[[:space:]]*[0-9]+' "$RESS_TMP")
if [ "$SEQ_COUNT" -ge 3 ]; then
  pass "RESS-02: sequence_number present on at least 3 events (captured ${SEQ_COUNT})"
else
  fail "RESS-02: sequence_number found on only ${SEQ_COUNT} events (expected ≥3)"
fi

# P3-04: heartbeat must be SSE comment, not data event.
if ! grep -qE '^data:.*heartbeat' "$RESS_TMP"; then
  pass "P3-04: no 'heartbeat' string in data: lines"
else
  fail "P3-04: heartbeat leaked as data: event (regression — see PITFALLS.md §P3-04)"
fi

# RESS-04: Responses API uses response.completed as terminator, NEVER [DONE].
if ! grep -qF 'data: [DONE]' "$RESS_TMP"; then
  pass "RESS-04: no [DONE] terminator (Responses uses response.completed only)"
else
  fail "RESS-04: stray [DONE] in /v1/responses stream"
fi

echo "[smoke-test-router] --- Phase 16 RESS section complete ---"
```
