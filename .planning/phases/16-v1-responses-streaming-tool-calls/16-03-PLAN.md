**SSE-parsing helper:**

```ts
function parseSse(body: string): { event?: string; data?: unknown }[] {
  return body.split('\n\n').filter(Boolean).map((block) =&gt; {
    const lines = block.split('\n');
    const event = lines.find((l) =&gt; l.startsWith('event: '))?.slice(7);
    const dataLine = lines.find((l) =&gt; l.startsWith('data: '))?.slice(6);
    const comment = lines.find((l) =&gt; l.startsWith(': '));
    return { event, data: dataLine ? JSON.parse(dataLine) : undefined, comment };
  });
}
```
