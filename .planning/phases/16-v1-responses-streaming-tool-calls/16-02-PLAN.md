import { beforeAll, afterAll, vi } from 'vitest';

describe('canonicalToResponsesSse — golden fixtures (RESS-04)', () =&gt; {
  beforeAll(() =&gt; {
    vi.useFakeTimers({ shouldAdvanceTime: false });
    vi.setSystemTime(new Date(0));

  for (const file of files) {
    it(`golden: ${file}`, async () =&gt; {
      const fx = JSON.parse(readFileSync(`${dir}${file}`, 'utf8')) as {
        canonical_events: unknown[];
        expected_sse: { event: string; data: unknown }[];
        opts?: CanonicalToResponsesSseOpts;
      };
      const events = fx.canonical_events as CanonicalStreamEvent[];
      // Convert the __throw__ sentinel to a throwing generator for fixture #5.
      async function* gen() {
        for (const e of events) {
          if ((e as { type?: string }).type === '__throw__') throw new Error('upstream-failed');
          yield e as CanonicalStreamEvent;
        }
      }
      const out = await collect(canonicalToResponsesSse(gen(), fx.opts));
      const actual = out.map((f) =&gt; ({ event: f.event, data: JSON.parse(f.data) as unknown }));
      if (process.env.UPDATE_GOLDEN === '1') {
        writeFileSync(`${dir}${file}`, JSON.stringify({ ...fx, expected_sse: actual }, null, 2) + '\n');
      } else {
        expect(actual).toEqual(fx.expected_sse);
      }
    });
  }
});
