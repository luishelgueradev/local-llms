**Step 1 — Create `router/src/backends/liveness.ts`:**

Verbatim from 03-PATTERNS.md §liveness.ts (lines 227–268) + 03-RESEARCH §Pattern 2 (lines 286–375). Combined target shape:

```ts
import type { Logger } from 'pino';

export type ProbeStatus = 'alive' | 'down' | 'stale';

export interface ProbeResult {
  status: 'alive' | 'down';
  lastProbeAt: string;
  latencyMs?: number;
  error?: string;
}

export interface LivenessScheduler {
  get(url: string): ProbeResult | undefined;
  urls(): string[];
  start(urls: string[]): void;
  stop(): void;
  refresh(): Promise<void>;
}

export interface MakeLivenessSchedulerOpts {
  intervalMs?: number;
  timeoutMs?: number;
  probe: (url: string, signal: AbortSignal) => Promise<{ ok: boolean; latencyMs: number; error?: string }>;
  logger: Pick<Logger, 'info' | 'warn' | 'error' | 'debug'>;
}

export function makeLivenessScheduler(opts: MakeLivenessSchedulerOpts): LivenessScheduler {
  const intervalMs = opts.intervalMs ?? 10_000;
  const timeoutMs = opts.timeoutMs ?? 2_000;
  const timers = new Map<string, NodeJS.Timeout>();
  const cache = new Map<string, ProbeResult>();
  const inFlight = new Set<string>();   // A9 guard
  let stopped = false;

  const runOne = async (url: string): Promise<void> => {
    if (stopped) return;
    if (inFlight.has(url)) return;        // A9: skip if already probing
    inFlight.add(url);
    const ac = new AbortController();
    const t = setTimeout(() => ac.abort(new Error('probe-timeout')), timeoutMs);
    const previous = cache.get(url)?.status;
    try {
      const { ok, latencyMs, error } = await opts.probe(url, ac.signal);
      const next: ProbeResult = ok
        ? { status: 'alive', lastProbeAt: new Date().toISOString(), latencyMs }
        : { status: 'down',  lastProbeAt: new Date().toISOString(), latencyMs, error };
      cache.set(url, next);
      if (previous !== undefined && previous !== next.status) {
        opts.logger.info(
          { event: 'backend_liveness', url, previous, current: next.status, latencyMs },
          'liveness transition',
        );
      } else if (next.status === 'down') {
        opts.logger.debug({ url, error }, 'liveness probe down (sustained)');
      }
    } finally {
      clearTimeout(t);
      inFlight.delete(url);
    }
  };

  return {
    get: (url) => cache.get(url),
    urls: () => Array.from(timers.keys()),
    start(urls: string[]) {
      const distinct = Array.from(new Set(urls));
      // Clear timers for URLs no longer present
      for (const url of Array.from(timers.keys())) {
        if (!distinct.includes(url)) {
          clearInterval(timers.get(url)!);
          timers.delete(url);
          cache.delete(url);
        }
      }
      // Add timers for new URLs
      for (const url of distinct) {
        if (timers.has(url)) continue;
        timers.set(url, setInterval(() => void runOne(url), intervalMs));
        void runOne(url);   // immediate first probe
      }
    },
    stop() {
      if (stopped) return;
      stopped = true;
      for (const [, t] of timers) clearInterval(t);
      timers.clear();
      // Note: cache is NOT cleared on stop; tests inspecting state-after-stop are OK
    },
    async refresh() {
      await Promise.all(Array.from(timers.keys()).map(runOne));
    },
  };
}
```

**Critical implementation details (each is a test case in Step 2):**
- `start(urls)` MUST be idempotent (de-dup against `timers` Map)
- `start(urls)` MUST remove timers for URLs no longer in the new set (so hot-reload to a smaller registry doesn't leak)
- `runOne` MUST skip if `inFlight.has(url)` (A9 guard)
- `stop()` MUST be idempotent (`stopped` flag)
- Transition log at `info`, sustained-down at `debug`
- Immediate first probe — call `runOne(url)` once synchronously (well, fire-and-forget) before the interval tick

Cases:

1. **Immediate first probe**: `start(['urlA'])` → assert `probe` was called with `'urlA'` synchronously (or within microtask flush via `await vi.advanceTimersByTime(0)`).

2. **Interval-driven probes**: Set `intervalMs: 100`; `start(['urlA'])`; await first probe; advance time by 300ms; assert `probe` was called ≥ 3 times for `'urlA'` (immediate + 2 intervals; exact count tolerates ±1 for timing slop).

3. **Cache populated on success**: `probe` resolves `{ok: true, latencyMs: 5}`; after immediate probe, `scheduler.get('urlA')` returns `{ status: 'alive', lastProbeAt: <iso>, latencyMs: 5 }`.

4. **Cache populated on failure**: `probe` resolves `{ok: false, latencyMs: 100, error: 'ECONNREFUSED'}`; after immediate probe, `get('urlA')` returns `{ status: 'down', lastProbeAt, latencyMs: 100, error: 'ECONNREFUSED' }`.

5. **De-dup on repeated start (Pitfall 6)**: `start(['A', 'B'])`; `start(['A', 'B'])` again; assert `scheduler.urls()` length is 2 (not 4); assert that after one intervalMs has elapsed, `probe` was called approximately 4 times total (immediate × 2 from each start + 1 interval tick × 2) — NOT 8 times. (Exact count: 2 immediate + 2 interval ticks = 4. The second `start()` call should be a no-op for already-scheduled URLs, but the `void runOne(url)` line is inside the `for...continue` block — verify the implementation skips the immediate run when the timer is already scheduled. Adjust the test count to match: 2 immediate from first start + 0 from second start + 2 from first interval = 4.)

6. **URL set shrinkage**: `start(['A', 'B'])`; `start(['A'])` (B removed); assert `scheduler.urls()` is `['A']`; assert that no further calls to `probe('B')` occur after the shrink even when time advances.

7. **Transition log info**: probe alternates `{ok:true} → {ok:false}`; after the second probe, assert `logger.info` was called with first arg matching `{event:'backend_liveness', previous:'alive', current:'down'}`. Then alternate back; assert another info log with the inverse transition.

8. **Sustained-down at debug**: probe consistently `{ok:false}`; after the SECOND down probe (transition stable), assert `logger.info` was NOT called for that probe AND `logger.debug` WAS called.

9. **Overlapping-probe guard (A9)**: probe returns a never-resolving Promise on first call; advance time by `intervalMs * 3`; assert `probe` was called exactly ONCE for `'urlA'` (the in-flight one is held; subsequent ticks skip).

10. **stop() clears timers**: `start(['A'])`; advance time by intervalMs; assert probe was called twice; call `stop()`; advance time by `intervalMs * 5`; assert probe call count did NOT increase.

11. **stop() idempotent**: `stop(); stop(); stop()` does not throw and does not call `clearInterval` more than once on the same timer (use `vi.spyOn(global, 'clearInterval')` to verify call count).

12. **refresh()**: `start(['A', 'B'])`; await immediate probes; reset `probe.mock.calls`; await `scheduler.refresh()`; assert probe was called once each for A and B.

**Step 3 — Create `router/src/routes/readyz.ts`:**

Verbatim from 03-RESEARCH §Code Examples lines 769–798 / 03-PATTERNS.md §readyz.ts lines 493–530:

```ts
import type { FastifyInstance } from 'fastify';
import type { RegistryStore } from '../config/registry.js';
import type { LivenessScheduler } from '../backends/liveness.js';

const STALE_FACTOR = 2;
const INTERVAL_MS = 10_000;   // must match the scheduler's intervalMs default

export function registerReadyz(
  app: FastifyInstance,
  registry: RegistryStore,
  liveness: LivenessScheduler,
): void {
  app.get('/readyz', async (req, reply) => {
    const now = Date.now();
    const distinctUrls = Array.from(new Set(registry.get().models.map((m) => m.backend_url)));
    const backends = distinctUrls.map((url) => {
      const r = liveness.get(url);
      if (!r) return { url, status: 'down' as const, error: 'never probed' };
      const age = now - new Date(r.lastProbeAt).getTime();
      const stale = age > STALE_FACTOR * INTERVAL_MS;
      return {
        url,
        status: stale ? ('stale' as const) : r.status,
        last_probe_at: r.lastProbeAt,
        latency_ms: r.latencyMs,
        ...(r.error ? { error: r.error } : {}),
      };
    });
    const allAlive = backends.length > 0 && backends.every((b) => b.status === 'alive');
    reply.code(allAlive ? 200 : 503);
    return {
      status: allAlive ? ('ready' as const) : ('not_ready' as const),
      checked_at: new Date(now).toISOString(),
      backends,
    };
  });
}
```

```ts
import { makeLivenessScheduler, type LivenessScheduler } from './backends/liveness.js';
import { makeAdapter as defaultMakeAdapter } from './backends/factory.js';   // already imported in Plan 02
import { registerReadyz } from './routes/readyz.js';

// inside buildApp, AFTER the registry/error-handler setup, BEFORE route registration:
//
// Adapter cache for probes (one adapter instance per distinct URL).
const probeAdapters = new Map<string, ReturnType<typeof defaultMakeAdapter>>();
const probeAdapterFor = (url: string) => {
  let a = probeAdapters.get(url);
  if (!a) {
    // We need a ModelEntry to build the adapter. Look it up from the registry by URL.
    const reg = opts.registry.get();
    const entry = reg.models.find((m) => m.backend_url === url);
    if (!entry) throw new Error(`No registry entry for URL "${url}"`);
    a = defaultMakeAdapter(entry);
    probeAdapters.set(url, a);
  }
  return a;
};

const liveness = makeLivenessScheduler({
  intervalMs: 10_000,
  timeoutMs: 2_000,
  logger: app.log,
  probe: async (url, signal) => {
    const adapter = probeAdapterFor(url);
    return adapter.probeLiveness(signal);
  },
});

// Decorate so index.ts can call liveness.start(urls) on hot-reload.
app.decorate('liveness', liveness);

// Kick off the first probe set against the current registry snapshot.
const distinctUrls = Array.from(new Set(opts.registry.get().models.map((m) => m.backend_url)));
liveness.start(distinctUrls);

// Shutdown hook (D-D7) — clears all timers so process exit is clean.
app.addHook('onClose', async () => {
  liveness.stop();
  probeAdapters.clear();
});

// Wire route AFTER liveness exists.
registerReadyz(app, opts.registry, liveness);
```
